import React, { useState, useEffect } from 'react';

const StatusBar = ({ user }) => {
  const [sysInfo, setSysInfo] = useState({
    env: 'CARGANDO...',
    dbHost: '...',
    dbName: '...'
  });
  const [now, setNow] = useState(new Date());

  // 1. Reloj en tiempo real
  useEffect(() => {
    const timer = setInterval(() => {
      setNow(new Date());
    }, 1000);
    return () => clearInterval(timer);
  }, []);

  // 2. Pedir info al backend de forma segura
  useEffect(() => {
    const fetchSysInfo = async () => {
      try {
        const response = await fetch(`${process.env.REACT_APP_API_URL || 'http://localhost:5000'}/api/system/version`);
        const result = await response.json();
        if (result.success) {
          setSysInfo({
            env: result.environment ? result.environment.toLowerCase() : 'local',
            dbHost: result.database.host || 'unknown',
            dbName: result.database.name || 'unknown'
          });
        }
      } catch (err) {
        console.error('Error recuperando info del sistema en StatusBar:', err);
      }
    };
    fetchSysInfo();
  }, []);

  // 3. Lógica de colores de texto (Solo cambia el color de la palabra AMBIENTE)
  const getEnvTextColor = (env) => {
    switch (env) {
      case 'production':
        return 'text-white'; // Blanco para Producción
      case 'development':
      case 'local':
      case 'dev':
        return 'text-cyan-400 font-bold'; // Celeste brillante para Desarrollo
      case 'test':
        return 'text-red-500 font-bold'; // Rojo para Test
      default:
        return 'text-red-500 font-bold'; // Rojo para cualquier otro caso (como pediste)
    }
  };

  const envColorClass = getEnvTextColor(sysInfo.env);
  const userName = user?.username || user?.email || 'elgaita';
  const userRole = user?.role || 'Admin';

  const fechaStr = now.toLocaleDateString('es-AR', { day: '2-digit', month: '2-digit', year: '2-digit' });
  const horaStr = now.toLocaleTimeString('es-AR', { hour: '2-digit', minute: '2-digit', second: '2-digit' });

  return (
    <footer className="fixed bottom-0 left-0 w-full bg-dark-800/95 backdrop-blur-sm border-t border-dark-700 py-2 text-center z-50">
      <p className="text-[11px] text-dark-300 font-mono tracking-wider">
        <span className={envColorClass}>
          AMBIENTE: {sysInfo.env.toUpperCase()}
        </span>
        {' '}| DB : : mongoose {sysInfo.dbHost} {sysInfo.dbName} | User : {userName} | Perfil : {userRole} | Fecha : {fechaStr} | Hora : {horaStr}
      </p>
    </footer>
  );
};

export default StatusBar;
